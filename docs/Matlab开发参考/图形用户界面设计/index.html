对话框设计​

  1. 公共对话框

    • 文件打开对话框

      [filename, pathname] = uigetfile({'*.jpg';'*.bmp';'*.gif'},'选择图片');

    • 文件保存对话框

      [sfilename, spathname] = uiputfile({'*.jpg';'*.bmp';'*.gif'},'保存图片','untitled.jpg');

  2. Matlab专用对话框

    • 错误信息对话框

      errordlg('输入错误,请重新输入', '错误信息')

    • 帮助对话框

      helpdlg('矩阵尺寸必须相等','在线帮助');

    • 信息提示对话框

      msgbox('你按了取消键', '保存失败', 'error');

    • 进程条

      % 建立进程条
      h = waitbar(0, 'pleas wait...');
      % 模拟进程条滑动
      for i = 1:10000
         waitbar(i/10000, h)
      end
      % 关闭进程条
      close(h)

    • 输入对话框

      prompt={'Input Name','Input Age'};
      title='Input Name and Age';
      lines=[2 1]';
      def={'John Smith','35'};
      answer=inputdlg(prompt,title,lines,def);

    • 警告信息对话框

      h=warndlg('数据类型不符', '警告');

Gui控件简介​

% --- Executes on button press in pushbutton1.
function pushbutton1_Callback(hObject, eventdata, handles)
% hObject    handle to pushbutton1 (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)
% 将handles结构里面的string转化成double型
s1=str2double(get(handles.edit1,'String'))
s2=str2double(get(handles.edit2,'String'))
% 进行两数相加并将结果放到静态文本框中
set(handles.text1,'String',s1+s2);
% --- Executes on button press in pushbutton1.
function pushbutton2_Callback(hObject, eventdata, handles)
% hObject    handle to pushbutton1 (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)
% 将handles结构里面的string转化成double型
s1=str2double(get(handles.edit1,'String'))
s2=str2double(get(handles.edit2,'String'))
% 进行两数相减并将结果放到静态文本框中
set(handles.text1,‘String’,s1-s2);

%% 打开文件的设计

% 定义一个全局变量
global x;
% 文件打开对话框
[filename, pathname] = uigetfile({'*.jpg';'*.bmp';'*.gif'},'选择图片');
if isequal(filename, 0) | isequal(pathname, 0)
    errordlg('没有打开图像', '出错');
    return;
else
    file = [pathname, filename];
    x = imread(file); % 读入图像
    % 设置显示的坐标轴
    axes(handles.axes1);
    % 显示图像
    imshow(x);
end
